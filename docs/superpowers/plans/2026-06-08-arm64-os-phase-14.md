# MyOSv2 Phase 14 — ELF64 loader + coreutils Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans. Checkbox steps.

**Goal:** Load ELF64 programs (map PT_LOAD segments with correct perms, zero .bss) and ship `/bin` coreutils the shell runs via exec.

**Architecture:** `elf.c` parses the ELF and calls `as_map_segment` (in `vm.c`) per `PT_LOAD`; `as_create_elf` builds a full address space (segments + stack) and returns the entry. Real program loading (`thread_create_image`, `proc_exec`) switches to ELF. `umain` returns an `int` exit status. The build produces several ELF programs embedded in the initrd.

Reference: `docs/superpowers/specs/2026-06-08-arm64-os-phase-14-design.md`.

```c
uint64_t *as_alloc_l0(void)            // a fresh top table sharing the kernel map
{
    uint64_t *l0 = alloc_table();
    l0[0] = mmu_kernel_l0_entry();
    return l0;
}

void as_map_segment(struct addrspace *as, uint64_t vaddr, const void *src,
                    uint64_t filesz, uint64_t memsz, int writable, int exec)
{
    uint64_t attr = ATTR_AF | ATTR_SH_INNER | ATTR_IDX_NORMAL | ATTR_NG;
    attr |= writable ? (AP_RW_ALL | ATTR_UXN | ATTR_PXN) : AP_RO_ALL;
    if (!exec && !writable) { attr |= ATTR_UXN | ATTR_PXN; }   // ro data: no-exec

    uint64_t vstart = vaddr & ~0xFFFUL;
    uint64_t vend   = (vaddr + memsz + PAGE - 1) & ~0xFFFUL;
    const uint8_t *s = (const uint8_t *)src;
    for (uint64_t pva = vstart; pva < vend; pva += PAGE) {
        uint64_t pa = (uint64_t)(uintptr_t)pmm_alloc();
        uint8_t *dst = (uint8_t *)(uintptr_t)pa;
        for (uint64_t i = 0; i < PAGE; i++) {
            uint64_t va = pva + i;
            dst[i] = (va >= vaddr && va < vaddr + filesz) ? s[va - vaddr] : 0;
        }
        map_page(as->l0, pva, pa, attr);
        page_incref(pa);
    }
}
```

```c
int elf_load(struct addrspace *as, const void *img, uint64_t len, uint64_t *entry)
{
    if (len < sizeof(Elf64_Ehdr)) { return -1; }
    const Elf64_Ehdr *eh = img;
    if (eh->e_ident[0]!=0x7f||eh->e_ident[1]!='E'||eh->e_ident[2]!='L'||eh->e_ident[3]!='F') return -1;
    if (eh->e_ident[4] != 2) { return -1; }       // ELFCLASS64
    if (eh->e_machine != 0xB7) { return -1; }      // EM_AARCH64
    *entry = eh->e_entry;
    const uint8_t *base = img;
    for (int i = 0; i < eh->e_phnum; i++) {
        const Elf64_Phdr *ph = (const Elf64_Phdr *)(base + eh->e_phoff + (uint64_t)i*eh->e_phentsize);
        if (ph->p_type != PT_LOAD) { continue; }
        as_map_segment(as, ph->p_vaddr, base + ph->p_offset, ph->p_filesz, ph->p_memsz,
                       (ph->p_flags & PF_W) ? 1 : 0, (ph->p_flags & PF_X) ? 1 : 0);
    }
    return 0;
}
```
**`as_create_elf` in `vm.c`** (declare in vm.h):
```c
struct addrspace *as_create_elf(const void *img, uint64_t len, uint64_t *entry)
{
    struct addrspace *as = (struct addrspace *)pmm_alloc();
    as->l0 = as_alloc_l0();
    as->asid = asid_alloc();
    if (elf_load(as, img, len, entry) != 0) { as_destroy(as); return 0; }
    uint64_t rw = ATTR_AF | ATTR_SH_INNER | ATTR_IDX_NORMAL | AP_RW_ALL | ATTR_UXN | ATTR_PXN | ATTR_NG;
    for (uint64_t i = 1; i <= 16; i++) {
        uint64_t pa = (uint64_t)(uintptr_t)pmm_alloc();
        map_page(as->l0, USER_STACK_TOP - i*PAGE, pa, rw);
        page_incref(pa);
    }
    return as;
}
```
