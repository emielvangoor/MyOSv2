# MyOSv2 Phase 12 — Graphics (ramfb framebuffer) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Allocate a 1280×720 XRGB8888 framebuffer, register it with QEMU's ramfb over fw_cfg, and draw a gradient + color bars + a text banner — with the shell still on the serial console.

**Architecture:** Four testable units — `bswap` helpers, framebuffer geometry + `ramfb_build_cfg`, drawing primitives, and the bitmap font — are built test-first against in-memory buffers. The hardware integration (`fwcfg.c` MMIO/DMA, `fb_init`, the kmain demo, the graphical `make run`) is wired last and confirmed visually, since the test build has no display.

**Tech Stack:** ARM64, C + inline asm MMIO, QEMU `virt` + `-device ramfb`, in-kernel KASSERT tests.

Reference: `docs/superpowers/specs/2026-06-08-arm64-os-phase-12-design.md`.

All fw_cfg values are **big-endian**; ARM is little-endian, so everything is byte-swapped.

```c
// fwcfg.h -- QEMU fw_cfg device: pass config blobs from QEMU to the guest.
// =======================================================================
//
// fw_cfg is a tiny MMIO device (base 0x09020000 on the virt board) that exposes
// named "items". We use it to register a ramfb framebuffer. EVERY multi-byte
// fw_cfg value is BIG-ENDIAN; ARM is little-endian, so we byte-swap in and out.
#pragma once
#include <stdint.h>

// Byte-swap helpers (host little-endian <-> fw_cfg big-endian).
static inline uint16_t bswap16(uint16_t v) { return (uint16_t)((v >> 8) | (v << 8)); }
static inline uint32_t bswap32(uint32_t v) { return __builtin_bswap32(v); }
static inline uint64_t bswap64(uint64_t v) { return __builtin_bswap64(v); }

// Look up an item's selector key by name (e.g. "etc/ramfb"); returns -1 if absent.
// On success, *size_out (if non-NULL) gets the item's byte size.
int  fwcfg_find_file(const char *name, uint32_t *size_out);

// DMA-transfer to/from a selected item. `control` carries the selector key in
// bits [31:16] plus SELECT and READ/WRITE bits. Returns 0 on success, -1 on error.
int  fwcfg_dma(uint32_t control, uint32_t length, void *buf);

// fw_cfg DMA control bits.
#define FWCFG_DMA_ERROR  0x01u
#define FWCFG_DMA_READ   0x02u
#define FWCFG_DMA_SKIP   0x04u
#define FWCFG_DMA_SELECT 0x08u
#define FWCFG_DMA_WRITE  0x10u
#define FWCFG_FILE_DIR   0x0019u
```

```c
// fb.h -- 1280x720 XRGB8888 framebuffer, scanned out by QEMU's ramfb device.
#pragma once
#include <stdint.h>

#define FB_WIDTH        1280
#define FB_HEIGHT       720
#define FB_BPP          4                              // bytes per pixel (XRGB8888)
#define FB_PITCH_PX     FB_WIDTH                        // pixels per row
#define FB_STRIDE_BYTES (FB_WIDTH * FB_BPP)             // bytes per row = 5120
#define FB_PAGES        ((FB_WIDTH * FB_HEIGHT * FB_BPP) / 4096)   // 900

#define DRM_FORMAT_XRGB8888 0x34325258u                // fourcc 'XR24'

// What the kernel draws into.
struct fb_info {
    volatile uint32_t *pixels;   // base of the framebuffer (identity-mapped RAM)
    uint32_t width, height;
    uint32_t pitch_px;           // pixels to advance one row
};

// The ramfb configuration blob handed to QEMU via fw_cfg. All fields big-endian.
struct ramfb_cfg {
    uint64_t addr;
    uint32_t fourcc;
    uint32_t flags;
    uint32_t width;
    uint32_t height;
    uint32_t stride;
} __attribute__((packed));

// Fill `cfg` (byte-swapped to big-endian) for a framebuffer at physical `addr`.
void ramfb_build_cfg(struct ramfb_cfg *cfg, uint64_t addr, uint32_t w, uint32_t h);

// Allocate the framebuffer and register it with ramfb. Returns 1 on success
// (and fills *out), 0 if ramfb is absent (no display). Never hangs.
int  fb_init(struct fb_info *out);
```

- [ ] **Step 4: Create `src/fb.c`** (the `ramfb_build_cfg` half; `fb_init` is added in Task 5)

```c
#include "fb.h"
#include "fwcfg.h"
#include "pmm.h"

void ramfb_build_cfg(struct ramfb_cfg *cfg, uint64_t addr, uint32_t w, uint32_t h)
{

    cfg->fourcc = bswap32(DRM_FORMAT_XRGB8888);
    cfg->flags  = 0;
    cfg->width  = bswap32(w);
    cfg->height = bswap32(h);
    cfg->stride = bswap32(w * FB_BPP);
}
```

```c
// draw.c -- pixel drawing primitives over a framebuffer.
#include <stdint.h>
#include "draw.h"
#include "font8x8.h"

uint32_t rgb(uint8_t r, uint8_t g, uint8_t b)
{
    return ((uint32_t)r << 16) | ((uint32_t)g << 8) | (uint32_t)b;
}

void draw_put(struct fb_info *f, int x, int y, uint32_t color)
{
    if (x < 0 || y < 0 || (uint32_t)x >= f->width || (uint32_t)y >= f->height) {
        return;                                  // off-screen -> ignore
    }
    f->pixels[(uint32_t)y * f->pitch_px + (uint32_t)x] = color;
}

void draw_fill_rect(struct fb_info *f, int x, int y, int w, int h, uint32_t color)
{
    for (int j = 0; j < h; j++) {
        for (int i = 0; i < w; i++) {
            draw_put(f, x + i, y + j, color);    // draw_put clips each pixel
        }
    }
}

void draw_clear(struct fb_info *f, uint32_t color)
{
    draw_fill_rect(f, 0, 0, (int)f->width, (int)f->height, color);
}

// Draw one 8x8 glyph scaled 2x (=> 16x16 cell, an 8x16-ish look). MSB = leftmost.
void draw_char(struct fb_info *f, int x, int y, char ch, uint32_t fg)
{
    const uint8_t *g = font_glyph(ch);
    for (int row = 0; row < 8; row++) {
        uint8_t bits = g[row];
        for (int col = 0; col < 8; col++) {
            if (bits & (0x80 >> col)) {
                // 2x scale: each font pixel -> a 2x2 block.
                draw_put(f, x + col * 2,     y + row * 2,     fg);
                draw_put(f, x + col * 2 + 1, y + row * 2,     fg);
                draw_put(f, x + col * 2,     y + row * 2 + 1, fg);
                draw_put(f, x + col * 2 + 1, y + row * 2 + 1, fg);
            }
        }
    }
}

void draw_text(struct fb_info *f, int x, int y, const char *s, uint32_t fg)
{
    int cx = x;
    for (; *s; s++) {
        if (*s == '\n') { cx = x; y += 18; continue; }
        draw_char(f, cx, y, *s, fg);
        cx += 18;                                // 16px glyph + 2px gap
    }
}
```

```c
// fwcfg.c -- QEMU fw_cfg MMIO + DMA driver (big-endian wire format).
#include <stdint.h>
#include "fwcfg.h"

#define FWCFG_BASE 0x09020000UL
#define FWCFG_DMA  (*(volatile uint64_t *)(FWCFG_BASE + 0x10))   // DMA addr register

// The control block the device reads; all fields big-endian.
struct fwcfg_dma_access {
    uint32_t control;
    uint32_t length;
    uint64_t address;
} __attribute__((packed));

static volatile struct fwcfg_dma_access dma;   // in .bss == identity-mapped RAM

int fwcfg_dma(uint32_t control, uint32_t length, void *buf)
{
    dma.control = bswap32(control);
    dma.length  = bswap32(length);
    dma.address = bswap64((uint64_t)(uintptr_t)buf);
    __asm__ volatile("dsb sy" ::: "memory");          // publish the struct first
    FWCFG_DMA = bswap64((uint64_t)(uintptr_t)&dma);    // kick the transfer
    __asm__ volatile("dsb sy" ::: "memory");
    // QEMU completes synchronously; spin until the busy bits clear.
    while (bswap32(dma.control) & ~FWCFG_DMA_ERROR) { }
    return (bswap32(dma.control) & FWCFG_DMA_ERROR) ? -1 : 0;
}

// Minimal strcmp for fixed-length names.
static int name_eq(const char *a, const char *b)
{
    while (*a && *b) { if (*a != *b) { return 0; } a++; b++; }
    return *a == *b;
}

int fwcfg_find_file(const char *name, uint32_t *size_out)
{
    uint32_t count_be;
    // SELECT the file directory and read its 4-byte count.
    if (fwcfg_dma((FWCFG_FILE_DIR << 16) | FWCFG_DMA_SELECT | FWCFG_DMA_READ,
                  4, &count_be) != 0) {
        return -1;
    }
    uint32_t count = bswap32(count_be);

    struct { uint32_t size; uint16_t select; uint16_t reserved; char name[56]; }
        __attribute__((packed)) entry;
    for (uint32_t i = 0; i < count; i++) {
        // Sequential read (no SELECT): next 64-byte entry.
        if (fwcfg_dma(FWCFG_DMA_READ, sizeof(entry), &entry) != 0) { return -1; }
        if (name_eq(entry.name, name)) {
            if (size_out) { *size_out = bswap32(entry.size); }
            return (int)bswap16(entry.select);
        }
    }
    return -1;
}
```

```c
// (append to fb.c; add #include "kprintf.h")
int fb_init(struct fb_info *out)
{
    int key = fwcfg_find_file("etc/ramfb", 0);
    if (key < 0) {
        return 0;                          // no ramfb device -> no display
    }
    void *base = pmm_alloc_pages(FB_PAGES);
    if (!base) { return 0; }

    struct ramfb_cfg cfg;
    ramfb_build_cfg(&cfg, (uint64_t)(uintptr_t)base, FB_WIDTH, FB_HEIGHT);
    if (fwcfg_dma(((uint32_t)key << 16) | FWCFG_DMA_SELECT | FWCFG_DMA_WRITE,
                  sizeof(cfg), &cfg) != 0) {
        return 0;
    }

    out->pixels   = (volatile uint32_t *)base;
    out->width    = FB_WIDTH;
    out->height   = FB_HEIGHT;
    out->pitch_px = FB_PITCH_PX;
    return 1;
}
```
