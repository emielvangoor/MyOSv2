# MyOSv2 Phase 15 — user heap (sbrk + malloc) Plan

> REQUIRED SUB-SKILL: superpowers:executing-plans. Checkbox steps.

**Goal:** Grow a per-process heap on demand (`sbrk`) and provide user `malloc`/`free`.

**Architecture:** `struct addrspace` tracks `heap_base`/`heap_end`; `as_sbrk` maps demand-zeroed pages as the break grows. `SYS_SBRK` exposes it; `ulib` builds a first-fit `malloc`/`free` on top. A `/bin/mtest` program self-tests malloc live.

Reference: `docs/superpowers/specs/2026-06-08-arm64-os-phase-15-design.md`.

```c
uint64_t as_sbrk(struct addrspace *as, long incr)
{
    uint64_t old = as->heap_end;
    uint64_t neu = old + (uint64_t)incr;         // signed add (incr may be < 0)
    if (incr > 0) {
        uint64_t rw = ATTR_AF | ATTR_SH_INNER | ATTR_IDX_NORMAL | AP_RW_ALL |
                      ATTR_UXN | ATTR_PXN | ATTR_NG;
        uint64_t first = (old + PAGE - 1) & ~0xFFFUL;   // first page not yet mapped
        uint64_t last  = (neu + PAGE - 1) & ~0xFFFUL;   // page past the new break
        for (uint64_t va = first; va < last; va += PAGE) {
            uint64_t pa = (uint64_t)(uintptr_t)pmm_alloc();
            if (!pa) { return (uint64_t)-1; }
            uint8_t *p = (uint8_t *)(uintptr_t)pa;
            for (int i = 0; i < (int)PAGE; i++) { p[i] = 0; }   // demand-zero
            map_page(as->l0, va, pa, rw);
            page_incref(pa);
        }
    }
    as->heap_end = neu;
    return old;
}
```
Note: `old==heap_base==USER_HEAP_BASE` initially. The "first page not yet mapped"
is `ceil(old)`; since `heap_base` is page-aligned, the first grow maps starting at
`USER_HEAP_BASE`. Init `heap_base = heap_end = USER_HEAP_BASE;` in `as_create_elf`
and `as_create_image`.

```c
void *sys_sbrk(long incr) { return (void *)syscall3(SYS_SBRK, incr, 0, 0); }

// First-fit free list. Each block: [hdr | payload], hdr = { size, next }.
struct mhdr { unsigned long size; struct mhdr *next; };
static struct mhdr *freelist;

void *malloc(unsigned long n)
{
    n = (n + 15) & ~15UL;                       // 16-byte align the payload
    struct mhdr **pp = &freelist;
    for (struct mhdr *b = freelist; b; pp = &b->next, b = b->next) {
        if (b->size >= n) { *pp = b->next; return (void *)(b + 1); }
    }
    struct mhdr *b = sys_sbrk((long)(n + sizeof(struct mhdr)));
    if ((long)b == -1) { return 0; }
    b->size = n;
    return (void *)(b + 1);
}

void free(void *p)
{
    if (!p) { return; }
    struct mhdr *b = (struct mhdr *)p - 1;
    b->next = freelist;
    freelist = b;
}
```

```c
#include "ulib.h"
static void say(const char *s) { sys_write(1, s, ustrlen(s)); }
int umain(void)
{
    // 1. distinct blocks don't overlap
    char *a = malloc(32), *b = malloc(32);
    for (int i = 0; i < 32; i++) { a[i] = (char)i; b[i] = (char)(100 + i); }
    for (int i = 0; i < 32; i++) { if (a[i] != (char)i || b[i] != (char)(100+i)) { say("mtest: overlap FAIL\n"); return 1; } }
    // 2. free + reuse
    free(a);
    char *c = malloc(32);
    if (c != a) { say("mtest: reuse FAIL\n"); return 1; }
    // 3. big allocation (forces multi-page sbrk)
    char *big = malloc(9000);
    if (!big) { say("mtest: big FAIL\n"); return 1; }
    big[0] = 1; big[8999] = 2;
    if (big[0] != 1 || big[8999] != 2) { say("mtest: big rw FAIL\n"); return 1; }
    say("mtest: ok\n");
    return 0;
}
```
