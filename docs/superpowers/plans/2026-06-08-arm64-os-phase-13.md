# MyOSv2 Phase 13 — Process lifecycle (exec + exit + wait) Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Add `exec`, `exit(status)`, and `wait` with full address-space teardown and ASID recycling.

**Architecture:** Threads gain a `parent` and `exit_status` and a `ZOMBIE` state; `thread_exit(status)` zombifies and wakes a waiting parent; `sched_wait` reaps a zombie child (unlink + free stack + `as_destroy` + free struct). `as_destroy` walks the user page-table subtree, decrefs pages, frees tables, and recycles the ASID via a free list. `proc_exec` loads a file into a new address space and rewrites the trap frame to land in it.

**Tech Stack:** ARM64, C, in-kernel KASSERT tests via `make test`.

Reference: `docs/superpowers/specs/2026-06-08-arm64-os-phase-13-design.md`.

```c
#define ASID_FREE_MAX 256
static uint16_t asid_free_list[ASID_FREE_MAX];
static int      asid_free_n;

void asid_free(uint16_t a) {
    if (a != 0 && asid_free_n < ASID_FREE_MAX) { asid_free_list[asid_free_n++] = a; }
}

uint16_t asid_alloc(void) {
    if (asid_free_n > 0) { return asid_free_list[--asid_free_n]; }   // recycle first
    if (next_asid > ASID_MAX) { next_asid = 1; flush_all_tlb(); }
    return (uint16_t)(next_asid++);
}
```
In `vm_init()` add `asid_free_n = 0;`.

```c
void as_destroy(struct addrspace *as)
{
    uint64_t e1 = as->l0[1];                 // user region (l0[0] is shared kernel)
    if (e1 & 1) {
        uint64_t *l1 = (uint64_t *)(e1 & PA_MASK);
        for (int i = 0; i < 512; i++) {
            if (!(l1[i] & 1)) { continue; }
            uint64_t *l2 = (uint64_t *)(l1[i] & PA_MASK);
            for (int j = 0; j < 512; j++) {
                if (!(l2[j] & 1)) { continue; }
                uint64_t *l3 = (uint64_t *)(l2[j] & PA_MASK);
                for (int k = 0; k < 512; k++) {
                    if (l3[k] & 1) { page_decref(l3[k] & PA_MASK); }
                }
                pmm_free(l3);
            }
            pmm_free(l2);
        }
        pmm_free(l1);
    }
    pmm_free(as->l0);
    asid_free(as->asid);
    pmm_free(as);
}
```

- rewrite `thread_exit`:
```c
void thread_exit(int status)
{
    uint64_t flags = irq_save();
    current->exit_status = status;
    current->state = THREAD_ZOMBIE;
    if (current->parent && current->parent->state == THREAD_SLEEPING) {
        current->parent->state = THREAD_RUNNABLE;   // wake a waiter
    }
    schedule();
    irq_restore(flags);     // unreachable
    for (;;) { }
}
```
- add `sched_wait`:
```c
int sched_wait(int *status)
{
    for (;;) {
        uint64_t flags = irq_save();
        struct thread *prev = current, *child = 0, *any = 0;
        do {
            if (prev->next->parent == current) {
                any = prev->next;
                if (prev->next->state == THREAD_ZOMBIE) { child = prev->next; break; }
            }
            prev = prev->next;
        } while (prev != current);

        if (child) {
            if (status) { *status = child->exit_status; }
            int pid = child->id;
            prev->next = child->next;                 // unlink from the ring
            kfree(child->stack);
            if (child->as) { as_destroy(child->as); }
            kfree(child);
            irq_restore(flags);
            return pid;
        }
        if (!any) { irq_restore(flags); return -1; }  // no children at all
        current->state = THREAD_SLEEPING;             // wait for one to exit
        schedule();
        irq_restore(flags);
    }
}
```
Note: scanning via `prev->next` keeps the predecessor for unlinking. `schedule()`
already skips non-`RUNNABLE`.

Update `SYS_EXIT` later (Task 5). The existing `thread_exit()` call in
`thread_trampoline` (switch.S) passes no arg — change it to pass 0: in
`switch.S`, `mov x0, #0` before `bl thread_exit`. Add that.

```c
int proc_exec(struct trapframe *tf, const char *path)
{
    struct file *f = vfs_open(path);
    if (!f) { return -1; }
    uint64_t len = f->vnode->size;
    uint8_t *buf = kmalloc(len ? len : 1);
    f->off = 0;
    int n = vfs_read(f, buf, len);
    vfs_close(f);
    if (n <= 0) { kfree(buf); return -1; }

    struct addrspace *neu = as_create_image(buf, (uint64_t)n);
    kfree(buf);
    if (!neu) { return -1; }

    struct addrspace *old = sched_current_as();
    sched_set_current_as(neu);          // current->as = neu
    as_switch(neu);

    for (int i = 0; i < 31; i++) { tf->x[i] = 0; }
    tf->elr    = user_entry_va();
    tf->sp_el0 = USER_STACK_TOP;
    tf->spsr   = 0;                      // EL0t, IRQs on (matches enter_user)

    if (old) { as_destroy(old); }
    return 0;
}
```
Add `void sched_set_current_as(struct addrspace *as);` to `sched.h`/`sched.c`
(`current->as = as;`).

```c
static void run_external(const char *cmd)
{
    char path[64];
    int i = 0; const char *p = "/bin/";
    while (p[i]) { path[i] = p[i]; i++; }
    int j = 0; while (cmd[j] && i < 62) { path[i++] = cmd[j++]; }
    path[i] = 0;

    long pid = sys_fork();
    if (pid == 0) {                 // child
        sys_exec(path);             // returns only if it failed
        puts1("exec: not found\n");
        sys_exit(127);
    }
    int st = 0; sys_wait(&st);      // parent reaps
}
```
`spawn` builtin: `pid=fork(); if(child){ puts1("child here\n"); sys_exit(3);} int st; sys_wait(&st);` then print the status as a digit. Route unknown commands to `run_external`.
