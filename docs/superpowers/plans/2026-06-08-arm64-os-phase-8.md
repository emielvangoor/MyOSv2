# MyOSv2 — Phase 8 Implementation Plan (exec + file syscalls)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Load a program image from a file into a fresh address space and run it; add file descriptors + open/read/close syscalls.

**Architecture:** `as_create_image(img,len)` (a generalization of Phase 6b `as_create`) copies a program image into a private address space. `proc_spawn(path)` reads a file and runs it. Per-thread fd tables back `SYS_OPEN/READ/CLOSE` and fd-aware `SYS_WRITE`.

**Tech Stack:** Freestanding C, the Phase 6b VM, the Phase 7 VFS.

```c
// Build an address space whose code pages hold `img` (len bytes), mapped
// read-only EL0-executable at USER_CODE_VA; private stack + data as before.
struct addrspace *as_create_image(const void *img, uint64_t len)
{
    struct addrspace *as = (struct addrspace *)pmm_alloc();
    as->l0 = alloc_table();
    as->l0[0] = mmu_kernel_l0_entry();

    uint64_t code_attr = ATTR_AF | ATTR_SH_INNER | ATTR_IDX_NORMAL | AP_RO_ALL;
    uint64_t rw_attr   = ATTR_AF | ATTR_SH_INNER | ATTR_IDX_NORMAL | AP_RW_ALL |
                         ATTR_UXN | ATTR_PXN;

    // Private code pages: copy the image in.
    uint64_t npages = (len + PAGE - 1) / PAGE;
    if (npages == 0) { npages = 1; }
    for (uint64_t i = 0; i < npages; i++) {
        uint64_t pa = (uint64_t)(uintptr_t)pmm_alloc();
        uint8_t *dst = (uint8_t *)(uintptr_t)pa;
        const uint8_t *src = (const uint8_t *)img;
        for (uint64_t j = 0; j < PAGE; j++) {
            uint64_t k = i * PAGE + j;
            dst[j] = (k < len) ? src[k] : 0;
        }
        map_page(as->l0, USER_CODE_VA + i * PAGE, pa, code_attr);
    }

    for (uint64_t i = 1; i <= 16; i++) {
        uint64_t pa = (uint64_t)(uintptr_t)pmm_alloc();
        map_page(as->l0, USER_STACK_TOP - i * PAGE, pa, rw_attr);
    }
    uint64_t data_pa = (uint64_t)(uintptr_t)pmm_alloc();
    map_page(as->l0, USER_DATA_VA, data_pa, rw_attr);
    return as;
}

// Convenience: build an address space from the embedded .user program.
struct addrspace *as_create(void)
{
    extern char __user_start[];
    extern char __user_end[];
    return as_create_image(__user_start, (uint64_t)(__user_end - __user_start));
}
```

Remove the now-unused `vm_init`/`code_phys`/`code_pages`/the old per-AS body. Keep
`vm_init` as an empty stub (callers still invoke it):

```c
struct thread *thread_create_image(const void *img, uint64_t len, int priority)
{
    struct thread *t = kmalloc(sizeof(struct thread));
    uint8_t *kstack  = kmalloc(STACK_SIZE);
    uint8_t *c = (uint8_t *)&t->ctx;
    for (unsigned i = 0; i < sizeof(t->ctx); i++) { c[i] = 0; }
    t->stack = kstack; t->state = THREAD_RUNNABLE; t->id = next_id++;
    t->priority = priority; t->wake_tick = 0;
    for (int i = 0; i < 16; i++) { t->fds[i] = 0; }
    t->as = as_create_image(img, len);
    t->next = 0;
    t->ctx.sp  = (uint64_t)(uintptr_t)(kstack + STACK_SIZE);
    t->ctx.lr  = (uint64_t)(uintptr_t)user_entry_trampoline;
    t->ctx.x19 = user_entry_va();          // entry within the loaded image
    t->ctx.x20 = USER_STACK_TOP;
    uint64_t flags = irq_save();
    if (current) {
        struct thread *tail = current;
        while (tail->next != current) { tail = tail->next; }
        tail->next = t; t->next = current;
    } else { t->next = t; }
    irq_restore(flags);
    return t;
}
```

```c
// proc.c -- load and run a program from a file.
#include <stdint.h>
#include "proc.h"
#include "vfs.h"
#include "sched.h"
#include "kheap.h"

struct thread *proc_spawn(const char *path, int priority)
{
    struct file *f = vfs_open(path);
    if (!f) { return 0; }
    uint64_t len = f->vnode->size;
    uint8_t *buf = kmalloc(len ? len : 1);
    f->off = 0;
    int n = vfs_read(f, buf, len);
    vfs_close(f);
    if (n <= 0) { return 0; }
    return thread_create_image(buf, (uint64_t)n, priority);
}
```

- [ ] **Step 2: Implement in `src/syscall.c`**

Add `#include "vfs.h"`. Make `SYS_WRITE` fd-aware and add the new cases. Replace
the `SYS_WRITE` case and add the others:

```c
    case SYS_WRITE: {                       // x0=fd, x1=ptr, x2=len
        uint64_t fd = tf->x[0];
        const char *s = (const char *)(uintptr_t)tf->x[1];
        uint64_t len = tf->x[2];
        if (fd == 1 || fd == 2) {           // stdout/stderr -> UART
            for (uint64_t i = 0; i < len; i++) { uart_putc(s[i]); }
            ret = (long)len;
        } else {
            struct file **fds = sched_current_fds();
            if (fds && fd < 16 && fds[fd]) {
                ret = vfs_write(fds[fd], s, len);
            } else { ret = -1; }
        }
        break;
    }
    case SYS_OPEN: {                         // x0=path
        const char *path = (const char *)(uintptr_t)tf->x[0];
        struct file **fds = sched_current_fds();
        ret = -1;
        if (fds) {
            struct file *f = vfs_open(path);
            if (f) {
                for (int i = 3; i < 16; i++) {
                    if (!fds[i]) { fds[i] = f; ret = i; break; }
                }
                if (ret < 0) { vfs_close(f); }
            }
        }
        break;
    }
    case SYS_READ: {                         // x0=fd, x1=buf, x2=len
        uint64_t fd = tf->x[0];
        void *buf = (void *)(uintptr_t)tf->x[1];
        uint64_t len = tf->x[2];
        struct file **fds = sched_current_fds();
        if (fd == 0) { ret = 0; }            // stdin: nothing yet
        else if (fds && fd < 16 && fds[fd]) { ret = vfs_read(fds[fd], buf, len); }
        else { ret = -1; }
        break;
    }
    case SYS_CLOSE: {                         // x0=fd
        uint64_t fd = tf->x[0];
        struct file **fds = sched_current_fds();
        if (fds && fd < 16 && fds[fd]) { vfs_close(fds[fd]); fds[fd] = 0; ret = 0; }
        else { ret = -1; }
        break;
    }
```

## Task 4: User program does file I/O + spawn from `kmain`

**Files:** `src/user.c`, `src/kmain.c`

- [ ] **Step 1: Rewrite `src/user.c` to use strings + file I/O**

```c
#include <stdint.h>
#include "user.h"
#include "syscall.h"

#define USER __attribute__((section(".user")))

USER static long syscall(long n, long a0, long a1, long a2)
{
    register long x8 __asm__("x8") = n;
    register long x0 __asm__("x0") = a0;
    register long x1 __asm__("x1") = a1;
    register long x2 __asm__("x2") = a2;
    __asm__ volatile("svc #0" : "+r"(x0) : "r"(x8), "r"(x1), "r"(x2) : "memory");
    return x0;
}

USER static long ustrlen(const char *s) { long n = 0; while (s[n]) n++; return n; }

// String literals must live in .user too (one contiguous, relocatable blob).
USER static const char greeting[] = "loaded program says hello (EL0)\n";
USER static const char path[]     = "/motd";

USER void user_main(void)
{
    syscall(SYS_WRITE, 1, (long)(uintptr_t)greeting, ustrlen(greeting));

    long fd = syscall(SYS_OPEN, (long)(uintptr_t)path, 0, 0);
    if (fd >= 0) {
        char buf[64];
        long n = syscall(SYS_READ, fd, (long)(uintptr_t)buf, 63);
        if (n > 0) { syscall(SYS_WRITE, 1, (long)(uintptr_t)buf, n); }
        syscall(SYS_CLOSE, fd, 0, 0);
    }
    syscall(SYS_EXIT, 0, 0, 0);
}
```
